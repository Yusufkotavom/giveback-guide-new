### **Analisa Return on Investment:**

```python
# Website ROI Calculator
def calculate_website_roi(investment, monthly_visitors, conversion_rate, avg_order_value):
    """
    Calculate website ROI for business
    """
    # Monthly calculations
    monthly_conversions = monthly_visitors * (conversion_rate / 100)
    monthly_revenue = monthly_conversions * avg_order_value
    annual_revenue = monthly_revenue * 12
    
    # ROI calculation
    roi = ((annual_revenue - investment) / investment) * 100
    payback_months = investment / monthly_revenue
    
    return {
        'investment': investment,
        'monthly_visitors': monthly_visitors,
        'monthly_conversions': monthly_conversions,
        'monthly_revenue': monthly_revenue,
        'annual_revenue': annual_revenue,
        'roi': roi,
        'payback_months': payback_months
    }

# Example: Toko online dengan website Rp 15 juta
result = calculate_website_roi(
    investment=15_000_000,      # Rp 15 juta
    monthly_visitors=5000,      # 5000 visitors/bulan
    conversion_rate=2.5,        # 2.5% conversion rate
    avg_order_value=250_000     # Rp 250K average order
)

print(f"Website Investment: Rp {result['investment']:,}")
print(f"Monthly Revenue: Rp {result['monthly_revenue']:,}")
print(f"Annual Revenue: Rp {result['annual_revenue']:,}")
print(f"ROI: {result['roi']:.1f}%")
print(f"Payback Period: {result['payback_months']:.1f} months")
```
