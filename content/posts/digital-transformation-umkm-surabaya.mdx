```python
# Simple CRM implementation example
class UMKMCustomerManager:
    def __init__(self):
        self.customers = []
        self.interactions = []
        self.sales_pipeline = []
    
    def add_customer(self, customer_data):
        """Add new customer to database"""
        customer = {
            'id': self.generate_customer_id(),
            'name': customer_data['name'],
            'phone': customer_data['phone'],
            'email': customer_data.get('email', ''),
            'address': customer_data.get('address', ''),
            'source': customer_data.get('source', 'walk-in'),
            'first_purchase_date': None,
            'total_purchases': 0,
            'total_spent': 0,
            'last_interaction': datetime.now(),
            'preferences': [],
            'status': 'active'
        }
        
        self.customers.append(customer)
        return customer['id']
    
    def track_interaction(self, customer_id, interaction_type, details):
        """Track customer interactions"""
        interaction = {
            'customer_id': customer_id,
            'type': interaction_type,  # call, email, visit, purchase
            'details': details,
            'timestamp': datetime.now(),
            'staff_id': self.get_current_staff()
        }
        
        self.interactions.append(interaction)
        self.update_customer_last_interaction(customer_id)
    
    def calculate_customer_value(self, customer_id):
        """Calculate customer lifetime value"""
        customer = self.get_customer(customer_id)
        
        if not customer or customer['total_purchases'] == 0:
            return 0
        
        # Simple CLV calculation
        avg_order_value = customer['total_spent'] / customer['total_purchases']
        purchase_frequency = self.get_purchase_frequency(customer_id)
        customer_lifespan = self.estimate_lifespan(customer_id)
        
        clv = avg_order_value * purchase_frequency * customer_lifespan
        return clv
    
    def identify_at_risk_customers(self):
        """Identify customers who might churn"""
        at_risk = []
        
        for customer in self.customers:
            days_since_last_interaction = (datetime.now() - customer['last_interaction']).days
            
            # Customer hasn't interacted in 60 days
            if days_since_last_interaction > 60 and customer['total_purchases'] > 0:
                risk_score = min(days_since_last_interaction / 30, 10)  # Max score 10
                
                at_risk.append({
                    'customer': customer,
                    'risk_score': risk_score,
                    'recommended_action': self.get_retention_strategy(risk_score)
                })
        
        return sorted(at_risk, key=lambda x: x['risk_score'], reverse=True)
    
    def get_retention_strategy(self, risk_score):
        """Get retention strategy based on risk score"""
        if risk_score >= 8:
            return "Urgent: Personal call + special discount"
        elif risk_score >= 6:
            return "High: WhatsApp message + loyalty reward"
        elif risk_score >= 4:
            return "Medium: Email newsletter + product recommendation"
        else:
            return "Low: Social media engagement"
```

```python
def calculate_digital_transformation_roi(
    initial_investment,
    monthly_operational_savings,
    revenue_increase_percentage,
    current_monthly_revenue,
    implementation_months
):
    """
    Calculate ROI of digital transformation for UMKM
    """
    # Current annual revenue
    annual_revenue = current_monthly_revenue * 12
    
    # Revenue increase from digital transformation
    additional_annual_revenue = annual_revenue * (revenue_increase_percentage / 100)
    
    # Annual operational savings
    annual_operational_savings = monthly_operational_savings * 12
    
    # Total annual benefits
    total_annual_benefits = additional_annual_revenue + annual_operational_savings
    
    # Implementation costs (including ongoing costs)
    ongoing_annual_costs = initial_investment * 0.15  # 15% of initial investment
    total_annual_costs = ongoing_annual_costs
    
    # Net annual benefit
    net_annual_benefit = total_annual_benefits - total_annual_costs
    
    # ROI calculations
    simple_roi = ((net_annual_benefit - initial_investment) / initial_investment) * 100
    payback_period = initial_investment / (net_annual_benefit / 12)
    
    # 3-year projection
    three_year_benefits = net_annual_benefit * 3
    three_year_roi = ((three_year_benefits - initial_investment) / initial_investment) * 100
    
    return {
        'initial_investment': initial_investment,
        'additional_monthly_revenue': additional_annual_revenue / 12,
        'monthly_operational_savings': monthly_operational_savings,
        'annual_net_benefit': net_annual_benefit,
        'simple_roi': simple_roi,
        'payback_months': payback_period,
        'three_year_roi': three_year_roi,
        'break_even_timeline': f"{payback_period:.1f} months"
    }
```
