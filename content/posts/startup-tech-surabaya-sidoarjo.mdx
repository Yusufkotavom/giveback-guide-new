# Startup validation framework
class StartupValidationFramework:
    def __init__(self, startup_idea):
        self.idea = startup_idea
        self.validation_stages = ['problem', 'solution', 'market', 'business_model', 'scale']
        self.current_stage = 0
        self.metrics = {}
    
    def validate_problem(self):
        """Stage 1: Problem validation"""
        validation_criteria = {
            'problem_interviews': {'target': 100, 'threshold': 80},
            'pain_point_severity': {'target': 8, 'threshold': 7},  # 1-10 scale
            'willingness_to_pay': {'target': 0.7, 'threshold': 0.5},  # percentage
            'market_size': {'target': 1000000000, 'threshold': 100000000}  # IDR
        }
        
        return self.evaluate_criteria(validation_criteria)
    
    def validate_solution(self):
        """Stage 2: Solution validation"""
        validation_criteria = {
            'mvp_user_feedback': {'target': 4.5, 'threshold': 4.0},  # 1-5 scale
            'feature_usage_rate': {'target': 0.8, 'threshold': 0.6},
            'user_retention_week1': {'target': 0.6, 'threshold': 0.4},
            'nps_score': {'target': 50, 'threshold': 30}
        }
        
        return self.evaluate_criteria(validation_criteria)
    
    def validate_market(self):
        """Stage 3: Market validation"""
        validation_criteria = {
            'customer_acquisition_cost': {'target': 50000, 'threshold': 100000},  # IDR
            'lifetime_value': {'target': 500000, 'threshold': 200000},  # IDR
            'market_penetration': {'target': 0.01, 'threshold': 0.005},  # percentage
            'competitive_advantage': {'target': 3, 'threshold': 2}  # number of advantages
        }
        
        return self.evaluate_criteria(validation_criteria)
    
    def validate_business_model(self):
        """Stage 4: Business model validation"""
        validation_criteria = {
            'monthly_recurring_revenue': {'target': 100000000, 'threshold': 25000000},  # IDR
            'gross_margin': {'target': 0.7, 'threshold': 0.4},
            'payback_period': {'target': 12, 'threshold': 24},  # months
            'revenue_growth_rate': {'target': 0.2, 'threshold': 0.1}  # monthly
        }
        
        return self.evaluate_criteria(validation_criteria)
    
    def validate_scale(self):
        """Stage 5: Scale validation"""
        validation_criteria = {
            'unit_economics_positive': {'target': True, 'threshold': True},
            'operational_efficiency': {'target': 0.9, 'threshold': 0.7},
            'team_scalability': {'target': 50, 'threshold': 20},  # team size
            'funding_readiness': {'target': True, 'threshold': True}
        }
        
        return self.evaluate_criteria(validation_criteria)
    
    def evaluate_criteria(self, criteria):
        """Evaluate validation criteria"""
        passed_criteria = 0
        total_criteria = len(criteria)
        
        for criterion, values in criteria.items():
            if criterion in self.metrics:
                actual_value = self.metrics[criterion]
                threshold = values['threshold']
                
                if isinstance(threshold, bool):
                    passed = actual_value == threshold
                elif isinstance(threshold, (int, float)):
                    passed = actual_value >= threshold
                else:
                    passed = False
                
                if passed:
                    passed_criteria += 1
        
        validation_score = passed_criteria / total_criteria
        return {
            'stage': self.validation_stages[self.current_stage],
            'score': validation_score,
            'passed': validation_score >= 0.8,  # 80% threshold
            'next_actions': self.get_next_actions(validation_score)
        }
    
    def get_next_actions(self, score):
        """Get recommended next actions based on validation score"""
        if score >= 0.8:
            return ["Move to next validation stage", "Document learnings", "Prepare for scaling"]
        elif score >= 0.6:
            return ["Improve weak metrics", "Gather more data", "Iterate on solution"]
        else:
            return ["Pivot or iterate significantly", "Re-validate core assumptions", "Consider new approach"]

# Usage example

startup_framework.metrics = {
    'problem_interviews': 95,

}

print(f"Next Actions: {validation_result['next_actions']}")
