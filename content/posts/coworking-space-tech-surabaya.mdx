def calculate_coworking_space_economics(capacity, location_tier):
    """Calculate coworking space financial model"""
    
    # Monthly operating costs
    operating_costs = {
        'rent': 150_000 * capacity,  # Rp 150K per desk equivalent
        'utilities': 50_000 * capacity,  # Electricity, internet, water
        'staff': 25_000_000,  # Management, community, cleaning
        'technology': 15_000_000,  # Software licenses, IT maintenance
        'marketing': 10_000_000,  # Digital marketing, events
        'insurance': 5_000_000,  # Property, liability insurance
        'maintenance': 8_000_000,  # Facility maintenance, repairs
        'supplies': 3_000_000,  # Office supplies, refreshments
    }
    
    total_monthly_costs = sum(operating_costs.values())
    
    # Revenue calculations
    occupancy_rate = 0.75  # 75% average occupancy
    average_revenue_per_desk = 2_000_000  # Rp 2M per desk per month
    
    monthly_revenue = capacity * occupancy_rate * average_revenue_per_desk
    monthly_profit = monthly_revenue - total_monthly_costs
    
    # ROI calculations
    initial_investment = capacity * 15_000_000  # Rp 15M setup per desk
    annual_profit = monthly_profit * 12
    roi_percentage = (annual_profit / initial_investment) * 100
    payback_period = initial_investment / monthly_profit
    
    return {
        'capacity': capacity,
        'monthly_costs': total_monthly_costs,
        'monthly_revenue': monthly_revenue,
        'monthly_profit': monthly_profit,
        'annual_profit': annual_profit,
        'initial_investment': initial_investment,
        'roi_percentage': roi_percentage,
        'payback_months': payback_period,
        'break_even_occupancy': total_monthly_costs / (capacity * average_revenue_per_desk)
    }

# Example: 100-desk coworking space
